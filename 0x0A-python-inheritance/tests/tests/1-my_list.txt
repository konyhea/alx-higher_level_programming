#!/usr/bin/python3

import unittest
from mylist import MyList  # Assuming your MyList class is defined in a separate module called "mylist"

class TestMyList(unittest.TestCase):
    def test_print_sorted_success(self):
        # Create an instance of MyList
        my_list = MyList([3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5])

        # Call the print_sorted method
        my_list.print_sorted()

        # No assertion needed for success scenario

    def test_print_sorted_failure(self):
        # Create an instance of MyList with non-integer elements
        my_list = MyList(["apple", "banana", "cherry"])

        # Call the print_sorted method
        my_list.print_sorted()

        # Assert that the list remains unchanged (since it cannot be sorted)
        self.assertEqual(my_list, ["apple", "banana", "cherry"])

if __name__ == "__main__":
    unittest.main()
